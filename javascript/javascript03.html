01. 선언적 함수

함수를 선언할 때 이름을 붙여주는 함수이다. 

function func() {
    document.write("함수가 실행되었습니다.1
");
}
func();     // 이게 있어야 출력됨

02. 익명 함수

말 그대로 이름 없는 함수로 앞에 const를 붙이며 시작하고 함수코드가 변수명에 저장된 형태이다. 

const func = function() {        // const func()라는 변수명에 함수 코드를 저장시키는 것
    document.write("함수가 실행되었습니다.2
");
}
func();

03. 매개변수 함수

함수 호출 시 인수로 전달된 값을 함수 내부에서 사용 가능하도록 해주는 변수로 변수는 여러 개 사용도 가능하다. 

function func(str1, str2, str3) {        // 변수 여러개 저장 가능
    document.write(str1);
    document.write(str2);
    document.write(str3);
}
func("함수가 실행되었습니다.3", "함수실행.3", "함수.3");        // 순서대로 func(str1, str2, str3...)이런 식으로 출력되는 것

04. 리턴값 함수

함수가 기능을 수행하고 그 결과를 호출한 곳으로 돌려주는(return) 값이다. 

function func() {
    const str = "함수가 실행되었습니다.4
";
    return str;     // return으로 결과값 뿌려줌, 종료를 의미하기도 함.
}
document.write(func());
결과보기
함수가 실행되었습니다.4

05. 화살표 함수 : 선언적 함수

함수를 선언할 때 이름을 붙여주는 함수이다.
아래 예시와 같이 화살표 식으로 간략화 할 수 있다. 최종적으로는 중괄호{ }도 생략 가능. 

// 화살표 함수로 간략화
/*
func = () => {
    document.write("함수가 실행되었습니다.5");
}
func();
*/

// 중괄호 생략
func = () => document.write("함수가 실행되었습니다.5"); 

func();
결과보기
"함수가 실행되었습니다.5"

06. 화살표 함수 : 익명 함수

말 그대로 이름 없는 함수로 앞에 const를 붙이며 시작하고 함수코드가 변수명에 저장된 형태이다.
아래 예시와 같이 화살표 식으로 간략화 할 수 있다. 최종적으로는 중괄호{ }도 생략 가능. 

/*
const func = () => {
    document.write("함수가 실행되었습니다.6");
}
*/

const func = () => document.write("함수가 실행되었습니다.6");

func();

07. 화살표 함수 : 매개변수 함수

함수 호출 시 인수로 전달된 값을 함수 내부에서 사용 가능하도록 해주는 변수로 변수는 여러 개 사용도 가능하다.
아래 예시와 같이 화살표 식으로 간략화 할 수 있다. 최종적으로는 중괄호{ }도 생략 가능. 

/*
func = (str) => {
    document.write(str);
}
*/
func = str => document.write(str);

func("함수가 실행되었습니다.7");

08. 화살표 함수 : 리턴값 함수

함수가 기능을 수행하고 그 결과를 호출한 곳으로 돌려주는(return) 값이다.
아래 예시와 같이 화살표 식으로 간략화 할 수 있다. 중괄호 안 내용이 두 줄이 되기 때문에 중괄호는 그대로 남겨둔다. 

func = () => {
    const str = "함수가 실행되었습니다.8";
    return str;
}

document.write(func());
결과보기
함수가 실행되었습니다.8

09. 화살표 함수 : 익명 함수 + 매개변수 + 리턴값

한 화살표 함수 안에 익명 함수와 매개변수 함수, 리턴값 함수를 모두 담은 예시. 

{
    const func = (str) => {     // 익명함수 = (매개변수) => {}
        return str;     // 리턴값
    }
    document.write(func("함수가 실행되었습니다.9
"));    // 리턴값처럼 출력
}
결과보기

10. 화살표 함수 : 익명 함수 + 매개변수 + 리턴값 + 괄호 생략

한 화살표 함수 안에 익명 함수와 매개변수 함수(변수가 1개일 경우에만 괄호 생략 可), 리턴값 함수를 모두 담은 예시. 

{
    const func = str => {   // 변수 1개일 때는 괄호 생략 가능
        return str;     // 결과 의미
    }
    document.write(func("함수가 실행되었습니다.10
"));
}

11. 화살표 함수 : 익명 함수 + 매개변수 + 리턴값 + 괄호 생략 + 리턴 생략

한 화살표 함수 안에 익명 함수와 매개변수 함수(변수가 1개일 경우에만 괄호 생략 可)를 모두 담고 리턴값 함수와 중괄호를 생략한 예시. 

{
    const func = str => str; // 리턴 생략, 중괄호도 생략

    document.write(func("함수가 실행되었습니다.11
"));
}

선언적 함수이므로 const를 쓰지 않는 점 외에는 11번과 같습니다.
